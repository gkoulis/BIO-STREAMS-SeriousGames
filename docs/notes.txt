Convert .webp to png.

```
for file in *.webp; do dwebp "$file" -o "${file%.webp}.png"; done
```

Remove BG
https://www.pixelcut.ai/t/background-remover

OR

```
mkdir -p output && for img in *.png; do convert "$img" -fuzz 20% -transparent white "output/$img"; done
mkdir -p output && for img in *.png; do convert "$img" -fuzz 15% -fill none -draw "color 0,0 floodfill" -alpha off -channel RGBA -alpha set "output/$img"; done
```
